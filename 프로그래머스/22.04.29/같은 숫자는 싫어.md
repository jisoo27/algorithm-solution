import java.util.*;

public class Solution {
     public static int[] solution(int[] arr) {
        Stack<Integer> stack = new Stack<>();
        stack.add(arr[0]);
        for (int i = 1; i < arr.length - 1; i++) {
            stack.add(arr[i]);
            if (arr[i - 1] == arr[i]) {
                stack.pop();
            }
        }
        int[] answer = stack.stream().mapToInt(Integer::intValue).toArray();
        return answer;
    }
}
