import java.util.*;

public class Solution {

    public static int solution(int n) {
        int sum = 0;
        List<Integer> list = new ArrayList<>();

        for (int i = 1; i <= n; i++) {
            if (n % i == 0) {
                list.add(i);
            }
        }
        for (Integer i : list) {
            sum += i;
        }
        return sum;
    }

        int n = sc.nextInt();
