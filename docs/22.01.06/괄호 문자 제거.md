import java.util.*;
class Main {
    public static String solution(String s){
        String answer = "";
        Stack<Character> stack = new Stack<>();
        for (char x : s.toCharArray()) {
            if (x == '(') {
                stack.push(x);
            } else if (x == ')') {
                stack.pop();
            } else if (stack.size() == 0 && x != '(' && x != ')') {

            }
        }
        return answer;
    }
