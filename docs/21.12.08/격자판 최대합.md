import java.util.*;

public class Main {

    public static int solution(int n, int[][] arr) {
        int answer = Integer.MIN_VALUE;
        int sum1; // í–‰ì˜ í•©
        int sum2; // ì—´ì˜ í•©
        for (int i = 0; i < n; i++) {

                sum1 += arr[i][j];
                sum2 += arr[j][i];
            }
            answer = Math.max(answer, sum1);
            answer = Math.max(answer, sum2);
        }
        sum1 = sum2 = 0;

    }

    public static void main(String[] args) {

        }
